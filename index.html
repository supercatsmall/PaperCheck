import tkinter as tk from tkinter import filedialog, messagebox, scrolledtext import base64 import os import threading def decrypt_data(encrypted_text): """五重解密：对应前端加密的逆操作""" # 1. Base64 解码 try: bytes_data = base64.b64decode(encrypted_text.strip()) except Exception: raise ValueError("Base64解码失败，文件可能已损坏") byte_list = list(bytes_data) # 2. 按位取反 (逆操作) for i in range(len(byte_list)): byte_list[i] = (~byte_list[i]) & 0xFF # 3. 减去37 (mod 256) for i in range(len(byte_list)): byte_list[i] = (byte_list[i] - 37) % 256 # 4. 字节反转 (逆操作) byte_list.reverse() # 5. XOR 0x5A (逆操作) for i in range(len(byte_list)): byte_list[i] ^= 0x5A return bytes(byte_list) class DecryptApp: def __init__(self, root): self.root = root self.root.title("文件还原工具") self.root.geometry("600x450") self.root.resizable(False, False) self.root.configure(bg="#f0f2f5") title = tk.Label(root, text="📂 IKUN 文件还原", font=("Segoe UI", 18, "bold"), fg="#1e293b", bg="#f0f2f5") title.pack(pady=20) desc = tk.Label(root, text="选择需要还原的 .ikun 文件或包含 .ikun 的文件夹", font=("Segoe UI", 10), fg="#64748b", bg="#f0f2f5") desc.pack() btn_frame = tk.Frame(root, bg="#f0f2f5") btn_frame.pack(pady=30) self.single_btn = tk.Button(btn_frame, text="📄 还原单个文件", font=("Segoe UI", 11, "bold"), bg="#3b82f6", fg="white", activebackground="#2563eb", relief="flat", padx=20, pady=10, cursor="hand2", command=self.decrypt_single_file) self.single_btn.grid(row=0, column=0, padx=10) self.folder_btn = tk.Button(btn_frame, text="📁 还原整个文件夹", font=("Segoe UI", 11, "bold"), bg="#10b981", fg="white", activebackground="#059669", relief="flat", padx=20, pady=10, cursor="hand2", command=self.decrypt_folder) self.folder_btn.grid(row=0, column=1, padx=10) log_label = tk.Label(root, text="操作日志：", font=("Segoe UI", 9, "bold"), fg="#475569", bg="#f0f2f5", anchor="w") log_label.pack(fill="x", padx=30, pady=(10, 0)) self.log_area = scrolledtext.ScrolledText(root, width=65, height=12, font=("Consolas", 9), bg="white", fg="#1e293b", relief="flat", borderwidth=1) self.log_area.pack(padx=30, pady=5) def log(self, message): self.log_area.insert(tk.END, message + "\n") self.log_area.see(tk.END) self.root.update() def decrypt_file(self, file_path): try: with open(file_path, 'r', encoding='utf-8') as f: content = f.read() original_bytes = decrypt_data(content) output_path = file_path[:-5] if file_path.endswith('.ikun') else file_path + '_restored' with open(output_path, 'wb') as f: f.write(original_bytes) return True, output_path except Exception as e: return False, str(e) def decrypt_single_file(self): file_path = filedialog.askopenfilename(title="选择 .ikun 文件", filetypes=[("IKUN 文件", "*.ikun"), ("所有文件", "*.*")]) if not file_path: return self.log(f"还原中: {file_path}") success, result = self.decrypt_file(file_path) if success: self.log(f"✅ 成功 → {result}") messagebox.showinfo("完成", f"文件已还原为:\n{result}") else: self.log(f"❌ 失败: {result}") messagebox.showerror("失败", f"无法还原:\n{result}") def decrypt_folder(self): folder_path = filedialog.askdirectory(title="选择包含 .ikun 文件的文件夹") if not folder_path: return self.log(f"扫描文件夹: {folder_path}") threading.Thread(target=self._decrypt_folder_worker, args=(folder_path,), daemon=True).start() def _decrypt_folder_worker(self, folder_path): count = 0 errors = [] for root, dirs, files in os.walk(folder_path): for file in files: if file.endswith('.ikun'): full_path = os.path.join(root, file) self.log(f"处理: {full_path}") success, result = self.decrypt_file(full_path) if success: self.log(f" ✅ {result}") count += 1 else: self.log(f" ❌ {result}") errors.append((full_path, result)) summary = f"\n🔍 完成，成功还原 {count} 个文件" if errors: summary += f"，{len(errors)} 个失败" self.log(summary) messagebox.showinfo("完成", summary) if __name__ == "__main__": root = tk.Tk() app = DecryptApp(root) root.mainloop()
